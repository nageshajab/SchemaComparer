IF NOT EXISTS (SELECT * FROM sys.columns WHERE object_id = OBJECT_ID(N'{1}') AND name = '{2}')
BEGIN
	if(@printErrorsOnly=1)	
		print 'column {2} missing in table {1}'
	else
	begin		
		ALTER TABLE {1}
		ADD {0};
	end
END
